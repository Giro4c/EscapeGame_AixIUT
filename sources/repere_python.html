    def leMotDePasseport (z)
        return ((z * 4 - 2) // 5)%3 + 10

        

    def leMotDePasserelle (y)
        laVar = y - 200
        laVarPasVar = y%3
        theVar = laVarPasVar * 10 + 31
        return (laVar + laVarPasVar + theVar)

        
    
    def leMotDePassage (x)
        uneVar = x - x%5
        autreVar = x + 25
        autreVar2 = autreVar // 2
        uneVar2 = (uneVar * 2) // 3
        return max({uneVar + uneVar2, autreVar + autreVar2})
